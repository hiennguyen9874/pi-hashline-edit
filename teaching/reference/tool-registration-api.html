Tool Registration API — Quick Reference

The interface between Pi extensions and the runtime

Extension entrypoint

// extensions/my-tool.ts
import type { ExtensionAPI } from "@earendil-works/pi-coding-agent";

export default function (pi: ExtensionAPI): void {
  pi.registerTool({ ... });
}

ToolDefinition shape

pi.registerTool({
  name:          string,        // unique id; replaces built-in if same name
  label:         string,        // UI display name
  description:   string,        // full description the LLM sees
  promptSnippet: string,        // one-line summary in tool list
  promptGuidelines?: string[],  // optional numbered rules (read only)
  parameters:    TObject,       // TypeBox schema (validates + documents args)
  renderShell?:  "default" | "self",

  renderCall(args, theme, ctx) → Text,     // live preview
  renderResult(result, meta, theme, ctx) → Text,  // post-execution rendering
  async execute(id, params, signal, onUpdate, ctx) → result,
});

Prompt text loading pattern

import { readFileSync } from "fs";

const DESC = readFileSync(

  "utf-8",
)
  .replaceAll("{{TEMPLATE_VAR}}", actualValue)

TypeBox schema pattern

import { Type } from "@sinclair/typebox";

const mySchema = Type.Object({
  path: Type.String({ description: "..." }),
  items: Type.Array(Type.String(), { description: "..." }),
}, { additionalProperties: false });

renderCall signature

renderCall(
  args: unknown,                          // current tool args (may be partial)
  theme: { bold, fg, ... },               // terminal styling functions
  context: {
    state: Record<string, unknown>,       // persists across renders
    cwd: string,                          // current working directory
    expanded: boolean,                    // user toggled expand
    executionStarted: boolean,            // execute() has been called
    argsComplete: boolean,                // all required args filled
    lastComponent: Text | undefined,      // reuse or replace
    invalidate(): void,                   // request re-render
  }
) → Text

renderResult signature

renderResult(
  result: unknown,                        // value returned by execute()
  meta: { isPartial: boolean },           // isPartial = streaming in progress
  theme: { fg, ... },
  context: {
    state, cwd, expanded,
    isError: boolean,                     // execute() threw
    lastComponent: Text | undefined,
    invalidate(): void,
  }
) → Text

execute signature

async execute(
  _toolCallId: string,                    // unique id for this invocation
  params: Infer<typeof schema>,          // validated args
  signal: AbortSignal,                    // cancel if user aborts
  _onUpdate: (update: unknown) => void,   // stream partial results
  ctx: ExtensionContext,                  // { cwd, ui, ... }
) → {
  content: Array<{ type: "text"; text: string }>,
  isError?: boolean,
  details?: any,                           // host-only metadata
}

Events API

// Emit
pi.events.emit("hashline:read-snapshot", { path, file, seenLines });

// Listen
pi.events.on("hashline:read-snapshot", (data) => { ... });

// Session lifecycle
pi.on("session_start", async (_event, ctx) => { ... });

File layout conventions

Directory Purpose
extensions/*.ts Entrypoints: default-export functions that wire src/ to the Pi runtime
src/*.ts Implementation: tool definitions, pipeline logic, utilities
tool-descriptions/*.md LLM-facing prompt text: full descriptions, snippets, guidelines
